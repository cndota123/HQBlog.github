正则表达式

正则表达式通常被用来检索、替换那些符合某个模式的文本。在多种语言中都支持正则表达式。

Js中正则表达式的使用

定义正则

  1. var re = new RegExp(“a”); //RegExp对象。参数就是我们想要制定的规则。有一种情况必须用这种方式，下面会提到。
  2. var re = /a/; // 简写方法 推荐使用 性能更好 不能为空 不然以为是注释

使用

  1. test() :在字符串中查找符合正则的内容，若查找到返回true,反之返回false.

function interweave(s1, s2) {
var re=/\d/;
if(s1==””){
return “”;
}
if(re.test(s1)){
var s3=s2.replace(re,””);
s1=s1.replace(re,s3);
return s1;
}
else{
var i=0,j=s2.length;
var s3=””;
for(;i<=j-1;i++){
s3=s3+s1.substr(i,1)+s2.substr(i,1);
}
s3=s3+s1.substring(i+1,s1.length-1);
return s3;
}
//code here
}
